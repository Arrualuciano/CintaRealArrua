33/**

47/**
48 * @brief Estados internos de la FSM de decodificacion de tramas UNER.
49 *
50 * @details La FSM sincroniza con el encabezado "UNER", luego captura el campo
51 * de longitud, el separador ':' y finalmente el payload byte a byte
52 * hasta verificar el checksum.
53 */
54typedef enum {
55 WAIT_U, /**< Esperando el caracter 'U' del encabezado. */
56 WAIT_N, /**< Esperando el caracter 'N'. */
57 WAIT_E, /**< Esperando el caracter 'E'. */
58 WAIT_R, /**< Esperando el caracter 'R'. */
59 WAIT_NBYTES, /**< Esperando el byte de longitud (LEN). */
60 WAIT_COLON, /**< Esperando el separador ':'. */
61 RECV_PAYLOAD /**< Recibiendo bytes del payload y checksum final. */
62} CommFsmState_t;

68void Comm_InitUART(void) {
69 UCSR0A = (1 << U2X0); /* Modo doble velocidad */
70 UBRR0 = 16; /* 115200 baud con U2X y F_CPU=16MHz */
71 UCSR0C = (3 << UCSZ00); /* 8 bits de datos, sin paridad, 1 stop */
72 UCSR0B = (1 << RXCIE0) | (1 << RXEN0) | (1 << TXEN0);
73}

104void Comm_Send(Comm_Protocol *comm, uint8_t cmdId, uint8_t *data, uint8_t nBytes) {
105 /* LEN = CMDID (1) + datos (nBytes) + CKS (1) */
106 uint8_t totalBytes = nBytes + 2;
107
108 /* Calcular checksum: XOR de todos los campos desde 'U' hasta el ultimo dato */
109 uint8_t cks = 'U' ^ 'N' ^ 'E' ^ 'R' ^ totalBytes ^ ':' ^ cmdId;
110 for (uint8_t i = 0; i < nBytes; i++) {
111 cks ^= data[i];
112 }
113
114 /* Encolar encabezado */
115 putTxByte(comm, 'U');
116 putTxByte(comm, 'N');
117 putTxByte(comm, 'E');
118 putTxByte(comm, 'R');
119
120 /* Encolar LEN y separador */
121 putTxByte(comm, totalBytes);
122 putTxByte(comm, ':');
123
124 /* Encolar CMDID y datos */
125 putTxByte(comm, cmdId);
126 for (uint8_t j = 0; j < nBytes; j++) {
127 putTxByte(comm, data[j]);
128 }
129
130 /* Encolar checksum */
131 putTxByte(comm, cks);
132}

134void Comm_Process(Comm_Protocol *comm) {
135 /* --- TX: transmitir un byte si hay datos pendientes y UART libre --- */
136 if (comm->txBuf.iw != comm->txBuf.ir) {
137 if (UCSR0A & _BV(UDRE0)) {
138 UDR0 = comm->txBuf.buf[comm->txBuf.ir++];
139 comm->txBuf.ir &= (comm->txBuf.size - 1);
140 }
141 }
142
143 /* --- RX: procesar todos los bytes disponibles con la FSM UNER --- */
144 while (comm->rxBuf.ir != comm->rxBuf.iw) {
145
146 uint8_t byte = comm->rxBuf.buf[comm->rxBuf.ir++];
147 comm->rxBuf.ir &= (comm->rxBuf.size - 1);
148
149 switch ((CommFsmState_t)comm->hdrState) {
150
151 case WAIT_U:
152 if (byte == 'U') comm->hdrState = WAIT_N;
153 /* Si no es 'U', permanecer en WAIT_U (resincronizacion automatica) */
154 break;
155
156 case WAIT_N:
157 if (byte == 'N') comm->hdrState = WAIT_E;
158 else comm->hdrState = WAIT_U;
159 break;
160
161 case WAIT_E:
162 if (byte == 'E') comm->hdrState = WAIT_R;
163 else comm->hdrState = WAIT_U;
164 break;
165
166 case WAIT_R:
167 if (byte == 'R') comm->hdrState = WAIT_NBYTES;
168 else comm->hdrState = WAIT_U;
169 break;
170
171 case WAIT_NBYTES:
172 /* Guardar LEN e inicializar el checksum acumulado */
173 comm->nBytes = byte;
174 comm->payloadIdx = 0;
175 comm->cks = 'U' ^ 'N' ^ 'E' ^ 'R' ^ byte ^ ':';
176 comm->hdrState = WAIT_COLON;
177 break;
178
179 case WAIT_COLON:
180 if (byte == ':') comm->hdrState = RECV_PAYLOAD;
181 else comm->hdrState = WAIT_U;
182 break;
183
184 case RECV_PAYLOAD:
185 /* nBytes: bytes restantes incluyendo CKS final
186 * Cuando nBytes == 1, el byte actual ES el checksum. */
187 if (comm->nBytes > 1) {
188
189 /* Proteccion de desbordamiento de buffer de payload */
190 if (comm->payloadIdx >= COMM_MAX_PAYLOAD) {
191 comm->hdrState = WAIT_U;
192 break;
193 }
194
195 comm->payload[comm->payloadIdx++] = byte;
196 comm->cks ^= byte;
197 comm->nBytes--;
198
199 } else {
200 /* Byte de checksum: validar e invocar callback */
201 if (comm->cks == byte) {
202 /* payload[0] = CMDID, &payload[1] = datos, payloadIdx-1 = nDatos */
203 comm->onCmd(comm->payload[0],
204 &comm->payload[1],
205 comm->payloadIdx - 1);
206 }
207 /* Reiniciar FSM para la proxima trama, valida o no */
208 comm->hdrState = WAIT_U;
209 }
210 break;
211 }
212 }
