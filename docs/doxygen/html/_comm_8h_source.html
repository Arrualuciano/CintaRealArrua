1/**
2 * @file Comm.h
3 * @brief Modulo de comunicacion serie con protocolo UNER.
4 *
5 * @details Implementa el protocolo de trama UNER sobre UART0 del ATmega.
6 * Formato de trama:
7 * @code
8 * [ 'U' | 'N' | 'E' | 'R' | LEN | ':' | CMDID | DATA... | CKS ]
9 * @endcode
10 * donde LEN = 1 (CMDID) + N (bytes de datos) + 1 (CKS), y
11 * CKS = XOR de todos los bytes desde 'U' hasta el ultimo dato.
12 *
13 * El modulo utiliza dos buffers circulares (RingBuf) de 128 bytes
14 * para desacoplar la ISR de recepcion del procesamiento en el
15 * main loop.
16 *
17 * @author Luciano
18 * @date Abril 2026
19 */
20

24/** @brief Tamano maximo del payload (CMDID + datos) que se puede recibir. */
25#define COMM_MAX_PAYLOAD 32

30/**
31 * @brief Tipo de callback invocado al recibir un comando valido.
32 *
33 * @param idComm Identificador del comando recibido.
34 * @param payload Puntero al arreglo de bytes de datos (sin CMDID ni CKS).
35 * @param nBytes Cantidad de bytes de datos en @p payload.
36 */
37typedef void (*comm_cmd_cb)(uint8_t idComm, uint8_t *payload, uint8_t nBytes);

39/**
40 * @brief Contexto completo de una instancia del protocolo UNER.
41 *
42 * @note Debe inicializarse con Comm_Init() antes de usar cualquier otra
43 * funcion del modulo.
44 */
45typedef struct {
46 RingBuf_t rxBuf; /**< Buffer circular de recepcion. */
47 RingBuf_t txBuf; /**< Buffer circular de transmision. */
48 comm_cmd_cb onCmd; /**< Callback para comandos validos. */
49 uint8_t cks; /**< Checksum acumulado durante recepcion. */
50 uint8_t timeOut; /**< Temporizador de timeout (reservado). */
51 uint8_t hdrState; /**< Estado actual de la FSM de decodificacion. */
52 uint8_t nBytes; /**< Bytes restantes del payload en curso. */
53 uint8_t payload[COMM_MAX_PAYLOAD]; /**< Buffer de payload en construccion. */
54 uint8_t payloadIdx; /**< Indice de escritura dentro de payload. */
55} Comm_Protocol;

84/**
85 * @brief Procesa el modulo de comunicacion: envio y recepcion.
86 *
87 * @details Debe llamarse periodicamente desde el main loop (no bloqueante).
88 * - TX: transmite un byte pendiente si el registro UDR0 esta libre.
89 * - RX: corre la FSM UNER sobre todos los bytes disponibles en el
90 * buffer de recepcion e invoca el callback @p onCmd al completar
91 * una trama valida con checksum correcto.
92 *
93 * @param comm Puntero a la instancia Comm_Protocol.

95void Comm_Process(Comm_Protocol *comm);

102 *
103 * @param comm Puntero a la instancia Comm_Protocol.
104 * @param cmdId Identificador del comando a enviar.
105 * @param data Puntero a los bytes de datos (puede ser NULL si nroBytes == 0).
106 * @param nroBytes Cantidad de bytes de datos en @p data.
107 */
108void Comm_Send(Comm_Protocol *comm, uint8_t cmdId, uint8_t *data, uint8_t nroBytes);
