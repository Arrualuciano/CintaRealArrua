1/**
2 * @file main.c
3 * @brief Aplicacion principal del clasificador de cajas en cinta transportadora real.
4 *
5 * @details Integra los modulos Comm, Actuator, Classifier, HCSR04 y Servo
6 * sobre un ATmega328P a 16 MHz.
7 *
8 * Arquitectura de tiempo:
9 * - Timer0 (prescaler 256): tick de 2 ms para FSMs y clasificador.
10 * - Timer1 (prescaler 8, free-running): base de tiempo en us para HC-SR04.
11 * - Timer2 (prescaler 64, CTC): PWM de 50 Hz para servomotores SG90.
12 *
13 * Pinout (segun consigna):
14 * - PB1: TRIGGER HC-SR04 (salida)
15 * - PB2: ECHO HC-SR04 (entrada)
16 * - PB3: SERVO 2 (salida)
17 * - PB4: SERVO 3 (salida)
18 * - PB5: LED heartbeat (salida)
19 * - PD5: IR sensor S0 - trigger zona medicion (entrada, pin 5)
20 * - PD2: IR sensor S1 - salida 0 (entrada, pin 2)
21 * - PD3: IR sensor S2 - salida 1 (entrada, pin 3)
22 * - PD4: IR sensor S3 - salida 2 (entrada, pin 4)
23 * - PC0: control de cinta transportadora (salida, activo-bajo)
24 * - PD7: SERVO 1 (salida)
25 * - USART0: comunicacion serie con HMI Qt (115200 baud, 8N1)
26 *
27 * Modos de operacion:
28 * - Normal (open-loop): S0 -> HC-SR04 -> Classifier -> Actuator.
29 * - Calibracion: actuadores deshabilitados, mide tiempos S0->S1/S2/S3.
30 *
31 * @author Luciano
32 * @date Mayo 2026
33 */

190/**
191 * @brief Despacha comandos recibidos via protocolo UNER desde el HMI.
192 *
193 * @details Soporta todos los comandos definidos en Protocol.h.
194 */
195static void OnCommCmd(uint8_t cmdId, uint8_t *payload, uint8_t nBytes) {
196 switch (cmdId) {
197
198 case CMD_ALIVE:
199 SendEvent(EVT_ALIVE, 0, 0);
200 break;
201
202 case CMD_START:
203 beltRunning = 1;
204 calibMode = 0;
205 PORTC &= ~(1 << PC0);
206 break;
207
208 case CMD_STOP:
209 beltRunning = 0;
210 PORTC |= (1 << PC0);
211 break;
212
213 case CMD_RESET:
214 beltRunning = 0;
215 calibMode = 0;
216 PORTC |= (1 << PC0);
217 Classifier_Reset(&classifier);
218 break;
219
220 case CMD_SET_CONFIG:
221 /* payload: [tipo0, tipo1, tipo2] */
222 if (nBytes >= 3) {
223 classifier.boxType[0] = payload[0];
224 classifier.boxType[1] = payload[1];
225 classifier.boxType[2] = payload[2];
226 }
227 break;
228
229 case CMD_SET_TIMES:
230 /* payload: [servo, timeHi, timeLo] */
231 if (nBytes >= 3) {
232 uint8_t servo = payload[0];
233 uint16_t ticks = ((uint16_t)payload[1] << 8) | payload[2];
234 if (servo < 3) {
235 classifier.timeArrival[servo] = ticks;
236 }
237 }
238 break;
239
240 case CMD_SET_THRESHOLD:
241 /* payload: [grande, mediana, pequena] */
242 if (nBytes >= 3) {
243 threshGrande = payload[0];
244 threshMediana = payload[1];
245 threshPequena = payload[2];
246 }
247 break;
248
249 case CMD_TEST_SERVO:
250 /* payload: [servoNum] - disparo manual de un servo */
251 if (nBytes >= 1) {
252 uint8_t servo = payload[0];
253 if (servo < 3) {
254 Actuator_Trigger(actuators[servo]);
255 }
256 }
257 break;
258
259 case CMD_SET_SERVO_ANGLE:
260 /* payload: [servoNum, pulseTicks] */
261 if (nBytes >= 2) {
262 Servo_SetPulse(payload[0], payload[1]);
263 }
264 break;
265
266 case CMD_CALIB_MODE:
267 /* payload: [1=entrar, 0=salir] */
268 if (nBytes >= 1) {
269 calibMode = payload[0];
270 if (calibMode) {
271 beltRunning = 0;
272 calibTickCounter = 0;
273 calibSent[0] = 0;
274 calibSent[1] = 0;
275 calibSent[2] = 0;
276 }
277 }
278 break;
279 }
280}

282/**
283 * @brief Registra una nueva caja y envia el evento al HMI.
284 * @param boxType Tipo de caja en cm (6, 8 o 10).
285 */
286static void OnNewBox(uint8_t boxType) {
287 Classifier_NewBox(&classifier, boxType);
288
289 /* Buscar a que servo fue asignada para informar al HMI */
290 for (uint8_t i = 0; i < 3; i++) {
291 if (classifier.boxType[i] == boxType) {
292 uint8_t evtData[4];
293 evtData[0] = boxType;
294 evtData[1] = i;
295 evtData[2] = (uint8_t)(classifier.timeArrival[i] >> 8);
296 evtData[3] = (uint8_t)(classifier.timeArrival[i] & 0xFF);
297 SendEvent(EVT_NEW_BOX, evtData, 4);
298 break;
299 }
300 }
301}

318/* =========================================================================
319 * Lectura de sensores IR (S1, S2, S3)
320 * ====================================================================== */
321
322/**
323 * @brief Lee los sensores IR S1 (PD2, pin 2), S2 (PD3, pin 3), S3 (PD4, pin 4) y envia eventos
324 * al HMI al detectar cambios de estado.
325 *
326 * @details En modo calibracion, al detectar flanco de subida de un sensor
327 * se envia el tiempo transcurrido desde S0 como EVT_CALIB_TIME.
328 */
329static void ProcessIRSensors(void) {
330 uint8_t pins[3];
331 pins[0] = (PIND & (1 << PD2)) ? 1 : 0; /* S1: pin 2 */
332 pins[1] = (PIND & (1 << PD3)) ? 1 : 0; /* S2: pin 3 */
333 pins[2] = (PIND & (1 << PD4)) ? 1 : 0; /* S3: pin 4 */
334
335 for (uint8_t i = 0; i < 3; i++) {
336 if (pins[i] != irPrev[i]) {
337 /* Enviar cambio de estado (sensor activo-bajo: invertir) */
338 uint8_t evtData[2];
339 evtData[0] = i + 1;
340 evtData[1] = !pins[i];
341 SendEvent(EVT_IR_SENSOR, evtData, 2);
342
343 /* Modo calibracion: flanco de bajada (activo-bajo) = objeto detectado */
344 if (calibMode && !pins[i] && !calibSent[i]) {
345 uint8_t calibData[3];
346 calibData[0] = i;
347 calibData[1] = (uint8_t)(calibTickCounter >> 8);
348 calibData[2] = (uint8_t)(calibTickCounter & 0xFF);
349 SendEvent(EVT_CALIB_TIME, calibData, 3);
350 calibSent[i] = 1;
351 }
352
353 irPrev[i] = pins[i];
354 }
355 }
356}

362/** Tarea de 2 ms: heartbeat LED, alive al HMI, contador de calibracion. */
363static void On2ms(void) {
364 GPIOR0 &= ~_BV(GPIOR00);
365
366 /* Heartbeat LED cada 100 ms */
367 if (--time100ms == 0) {
368 time100ms = 50;
369 PINB = (1 << PINB5);
370 }
371
372 /* Envio periodico de alive al HMI (~500 ms = 250 ticks) */
373 if (++aliveCounter >= 250) {
374 aliveCounter = 0;
375 SendEvent(EVT_ALIVE, 0, 0);
376 }
377
378 /* Decrementar debounce de S0 */
379 if (s0Debounce > 0) s0Debounce--;
380
381 /* Incrementar contador de calibracion si esta activo */
382 if (calibMode) {
383 calibTickCounter++;
384 }
385}

387/* =========================================================================
388 * Punto de entrada
389 * ====================================================================== */
390
391int main(void) {
392 cli();
393
394 time100ms = 50;
395
396 /* Configurar pines de I/O */
397 DDRB = (1 << PB5); /* PB5: LED heartbeat (salida) */
398 DDRB |= (1 << PB1); /* PB1: TRIGGER HC-SR04 (salida) */
399 DDRB &= ~(1 << PB2); /* PB2: ECHO HC-SR04 (entrada) */
400 DDRC |= (1 << PC0); /* PC0 (A0): control cinta (salida) */
401 PORTC |= (1 << PC0); /* Cinta apagada al inicio (activo-bajo) */
402 DDRD &= ~((1 << PD2) | (1 << PD3) | (1 << PD4) | (1 << PD5));
403 /* Pull-ups internos en los pines IR para evitar lecturas flotantes */
404 PORTD |= (1 << PD2) | (1 << PD3) | (1 << PD4) | (1 << PD5);
405
406 /* Tipos de salida por defecto: salida 0 = 6cm, salida 1 = 8cm, salida 2 = 10cm */
407 uint8_t defaultBoxType[3] = {6, 8, 10};
408
409 actuators[0] = &act0;
410 actuators[1] = &act1;
411 actuators[2] = &act2;
412
413 /* Inicializar perifericos y modulos */
414 InitTMR0();
415 InitTMR1();
416 Comm_Init(&comm, OnCommCmd);
417 HCSR04_Init(&sensor, hw_trigger_write, hw_echo_read, hw_get_us);
418 Servo_Init();
419 Actuator_Init(&act0, 0, 0, hw_Act_Set);
420 Actuator_Init(&act1, 1, 0, hw_Act_Set);
421 Actuator_Init(&act2, 2, 0, hw_Act_Set);
422 Classifier_Init(&classifier, defaultBoxType, OnActuatorTrigger);
423
424 sei();
425
426 /* ---- Super-loop ---- */
427 while (1) {
428
429 if (GPIOR0 & _BV(GPIOR00)) {
430 On2ms();
431 Actuator_Process(&act0);
432 Actuator_Process(&act1);
433 Actuator_Process(&act2);
434
435 /* Clasificador activo solo en modo normal con cinta en marcha */
436 if (beltRunning && !calibMode) {
437 Classifier_Process(&classifier);
438 }
439
440 /* Lectura de sensores IR cada 2 ms */
441 ProcessIRSensors();
442 }
443
444 Comm_Process(&comm);
445
446 /* S0 (PD5, pin 5): detectar flanco para disparar HC-SR04 (con debounce) */
447 {
448 uint8_t s0Now = (PIND & (1 << PD5)) ? 1 : 0;
449
450 if (s0Debounce == 0) {
451 if (s0Now != s0Prev) {
452 /* Notificar cambio de S0 al HMI (sensor activo-bajo: invertir) */
453 uint8_t evtData[2] = {0, !s0Now};
454 SendEvent(EVT_IR_SENSOR, evtData, 2);
455
456 if (!s0Now && s0Prev) {
457 /* Flanco de bajada (activo-bajo): objeto detectado */
458 if (sensor.state == HCSR_IDLE) {
459 sensor.state = HCSR_TRIGGER;
460 }
461
462 /* Resetear contador de calibracion al detectar caja en S0 */
463 if (calibMode) {
464 calibTickCounter = 0;
465 calibSent[0] = 0;
466 calibSent[1] = 0;
467 calibSent[2] = 0;
468 }
469
470 /* Iniciar debounce: ignorar rebotes durante 500 ms */
471 s0Debounce = S0_DEBOUNCE_TICKS;
472 }
473 }
474 }
475 s0Prev = s0Now;
476 }
477
478 HCSR04_Process(&sensor);
479
480 if (sensor.state == HCSR_READY) {
481 /* Enviar distancia medida al HMI */
482 {
483 uint8_t distData[2];
484 distData[0] = (uint8_t)(sensor.distance >> 8);
485 distData[1] = (uint8_t)(sensor.distance & 0xFF);
486 SendEvent(EVT_DISTANCE, distData, 2);
487 }
488
489 /* Clasificar segun umbrales (solo en modo normal) */
490 if (beltRunning && !calibMode) {
491 if (sensor.distance < threshGrande) {
492 OnNewBox(10);
493 } else if (sensor.distance < threshMediana) {
494 OnNewBox(8);
495 } else if (sensor.distance < threshPequena) {
496 OnNewBox(6);
497 }
498 }
499 sensor.state = HCSR_IDLE;
500 }
501
502 if (sensor.state == HCSR_ERROR) {
503 sensor.state = HCSR_IDLE;
504 }
505 }
506}
