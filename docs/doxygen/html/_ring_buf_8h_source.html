2 * @file RingBuf.h
3 * @brief Buffer circular (ring buffer) de proposito general.
4 *
5 * @details Estructura de datos para buffers circulares de tamano fijo.
6 * El tamano DEBE ser potencia de 2 para que el mascareo con
7 * (size - 1) funcione correctamente como reemplazo del modulo.
8 *
9 * Escritura (ISR o productor):
10 * @code
11 * buf[iw++] = dato;
12 * iw &= (size - 1);
13 * @endcode
14 *
15 * Lectura (main loop o consumidor):
16 * @code
17 * while (ir != iw) {
18 * uint8_t b = buf[ir++];
19 * ir &= (size - 1);
20 * }
21 * @endcode
22 *
23 * @note iw e ir son uint8_t, por lo que el tamano maximo util es 128
24 * (potencia de 2, menor que 256).
25 *
26 * @author Luciano
27 * @date Abril 2026
28 */
