1/**
2 * @file Actuator.h
3 * @brief Modulo de control de actuadores (pistones/brazos) de la cinta.
4 *
5 * @details Implementa una maquina de estados (FSM) no bloqueante para manejar
6 * el ciclo completo de un actuador mecanico:
7 *
8 * @code
9 * ACT_IDLE --> [Trigger()] --> ACT_WAITING --> ACT_EXTENDING --> ACT_RETRACTING --> ACT_IDLE
10 * @endcode
11 *
12 * - @b ACT_WAITING: Espera un retardo configurable antes de extender
13 * (permite sincronizar el disparo con la posicion
14 * fisica de la caja frente al brazo).
15 * - @b ACT_EXTENDING: Brazo extendido durante @p delayTicks ticks de 2 ms.
16 * - @b ACT_RETRACTING: Brazo retrayendose durante @p delayTicks ticks de 2 ms.
17 *
18 * Actuator_Process() debe llamarse cada 2 ms (tick del Timer0).
19 *
20 * @author Luciano
21 * @date Abril 2026
22 */
23

37/**
38 * @brief Estados internos de la FSM de cada actuador.
39 */
40typedef enum {
41 ACT_IDLE, /**< Brazo en reposo, listo para disparar. */
42 ACT_WAITING, /**< Esperando retardo de sincronizacion pre-extension. */
43 ACT_EXTENDING, /**< Brazo en proceso de extension. */
44 ACT_RETRACTING /**< Brazo en proceso de retraccion. */
45} ActuatorState;

47/**
48 * @brief Contexto de un actuador individual.

50typedef struct {
51 ActuatorState state; /**< Estado actual de la FSM. */
52 uint8_t armNum; /**< Identificador del brazo (0, 1 o 2). */
53 uint16_t timer; /**< Contador regresivo en ticks de 2 ms. */
54 uint16_t delayTicks; /**< Duracion de cada fase (waiting/extending/retracting) en ticks. */
55 actuator_hw_set_ptr SetActuator; /**< Funcion de hardware para mover el brazo. */
56} Actuator_t;

68/**
69 * @brief Solicita el disparo del actuador.
70 *
71 * @details Si el actuador esta en @p ACT_IDLE, inicia el ciclo de extension.

74 * @param act Puntero al actuador.

88void Actuator_Process(Actuator_t *act);
