5 * @details Cada caja medida recibe un servo destino y un timer de arribo.

9 * Invariantes de la lista:

20/**
21 * @brief Elimina la caja en la posicion @p idx desplazando las siguientes.
22 *
23 * @details Operacion O(n). Mantiene el orden de llegada de las cajas restantes.
24 *
25 * @param clas Puntero al clasificador.
26 * @param idx Indice (0-based) de la caja a eliminar.
27 */
28static void removeAt(classifier_t *clas, uint8_t idx) {
29 if (idx >= clas->count) return;
30 for (uint8_t i = idx; i + 1 < clas->count; i++) {
31 clas->list[i] = clas->list[i + 1];
32 }
33 clas->count--;
34}

40/**
41 * @brief Inicializa el clasificador con las asignaciones de salida.
42 *
43 * @details Resetea la lista de cajas en transito, guarda los tipos
44 * asignados a cada salida, el callback de disparo, y configura
45 * los tiempos de arribo por defecto para cada servo.
46 *
47 * @param clas Puntero al clasificador a inicializar.
48 * @param boxType Arreglo de CLASSIFIER_NUM_SENSORS elementos con el tipo

50 * @param onTrigger Callback que sera invocado cuando se deba disparar un actuador.
51 */
52void Classifier_Init(classifier_t *clas, uint8_t *boxType, classifier_trigger_cb onTrigger) {
53 for (uint8_t i = 0; i < CLASSIFIER_NUM_SENSORS; i++) {
54 clas->boxType[i] = boxType[i];
55 }
56 clas->count = 0;
57 clas->onTrigger = onTrigger;
58 clas->timeArrival[0] = TIME_ARRIVAL_SERVO0;
59 clas->timeArrival[1] = TIME_ARRIVAL_SERVO1;
60 clas->timeArrival[2] = TIME_ARRIVAL_SERVO2;
61}

71/**
72 * @brief Registra una nueva caja medida por el HC-SR04.
73 *
74 * @details Busca en boxType[] a que servo corresponde el tipo de caja,
75 * carga el timer de arribo configurado y agrega la caja a la lista.
76 * Si la lista esta llena o el tipo no coincide con ninguna salida,
77 * la caja se descarta silenciosamente.
78 *
79 * @param clas Puntero al clasificador.
80 * @param type Tipo de caja (altura en cm: 6, 8 o 10).
81 */
82void Classifier_NewBox(classifier_t *clas, uint8_t type) {
83 if (clas->count >= CLASSIFIER_MAX_BOXES) return; /* Lista llena: descartar */
84
85 /* Buscar a que servo va esta caja */
86 for (uint8_t i = 0; i < CLASSIFIER_NUM_SENSORS; i++) {
87 if (clas->boxType[i] == type) {
88 /* Encontro el servo: guardar datos y salir */
89 clas->list[clas->count].type = type;
90 clas->list[clas->count].servo = i;
91 clas->list[clas->count].timePushBox = clas->timeArrival[i];
92 clas->count++;
93 return;
94 }
95 }
96}

98/**
99 * @brief Decrementa los timers de las cajas en transito y dispara los servos.
100 *
101 * @details Debe llamarse cada 2 ms desde el tick del Timer0. Recorre la lista
102 * de cajas, decrementa timePushBox y al llegar a 0 invoca onTrigger()
103 * con el servo correspondiente. La caja disparada se elimina de la lista.
104 *
105 * @param clas Puntero al clasificador.
106 */
107void Classifier_Process(classifier_t *clas) {
108
109 for (uint8_t i = 0; i < clas->count; i++) {
110 clas->list[i].timePushBox--;
111 if (clas->list[i].timePushBox == 0) {
112 clas->onTrigger(clas->list[i].servo);
113 removeAt(clas, i);
114 i--; /* Compensar el desplazamiento de la lista */
115 }
116 }
