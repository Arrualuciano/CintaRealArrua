1/**
2 * @file HCSR04.c
3 * @brief Implementacion del driver para sensor ultrasonico HC-SR04.
4 *
5 * @details La FSM no bloqueante ejecuta la secuencia completa de medicion:
6 *
7 * 1. HCSR_TRIGGER: activa el pin TRIGGER en alto y registra t_ref.
8 * 2. HCSR_TRIG_WAIT: mantiene TRIGGER alto durante >= 10 us, luego
9 * lo baja y pasa a esperar el eco.
10 * 3. HCSR_ECHO_WH: espera el flanco ascendente en el pin ECHO.
11 * Si supera el timeout, pasa a HCSR_ERROR.
12 * 4. HCSR_ECHO_WL: mide la duracion del pulso ECHO hasta el flanco
13 * descendente y calcula la distancia en cm.
14 * 5. HCSR_READY: la distancia esta disponible en @p distance.
15 *
16 * Formula de conversion:
17 * @code
18 * distancia_cm = tiempo_echo_us / 58
19 * @endcode
20 * (velocidad del sonido: 343 m/s -> 29 us/cm ida, 58 us/cm ida+vuelta)
21 *
22 * @author Luciano
23 * @date Abril 2026
24 */
25

28/**
29 * @brief Inicializa el driver HC-SR04.
30 *
31 * @details Almacena los punteros a funcion para el acceso al hardware,
32 * configura el timeout por defecto (HCSR04_TIMEOUT_US) y deja
33 * el sensor en estado HCSR_IDLE con el pin TRIGGER en bajo.
34 *
35 * @param setting Puntero a la instancia HCSR04_Setting_t a inicializar.
36 * @param trigger Funcion de escritura del pin TRIGGER.
37 * @param echo Funcion de lectura del pin ECHO.
38 * @param timer Funcion que retorna el tiempo actual en microsegundos.
39 */
40void HCSR04_Init(HCSR04_Setting_t *setting, hcsr_write_ptr trigger,
41 hcsr_read_ptr echo, hcsr_get_time_ptr timer) {
42
43 /* Guardar punteros a funcion de hardware */
44 setting->trigger_write = trigger;
45 setting->echo_read = echo;
46 setting->get_us = timer;
47
48 /* Inicializar variables internas */
49 setting->state = HCSR_IDLE;
50 setting->t_ref = 0;
51 setting->timeout = HCSR04_TIMEOUT_US;
52 setting->distance = 0;
53
54 /* Asegurar que el pin TRIGGER comience en bajo */
55 setting->trigger_write(0);
56}

58/**
59 * @brief Ejecuta un paso de la FSM del sensor HC-SR04 (no bloqueante).
60 *
61 * @details Debe llamarse frecuentemente desde el super-loop. Para iniciar
62 * una nueva medicion, el usuario debe setear:
63 * @code
64 * setting->state = HCSR_TRIGGER;
65 * @endcode
66 *
67 * Al finalizar, el estado sera:
68 * - HCSR_READY: medicion exitosa, distancia en @p setting->distance (cm).
69 * - HCSR_ERROR: timeout, el sensor no respondio.
70 *
71 * El usuario debe resetear el estado a HCSR_IDLE despues de leer el
72 * resultado para permitir nuevas mediciones.
73 *
74 * @param setting Puntero a la instancia HCSR04_Setting_t.
75 */
76void HCSR04_Process(HCSR04_Setting_t *setting) {
77
78 switch (setting->state) {
79
80 case HCSR_IDLE:
81 /* Sensor inactivo, esperando solicitud externa */
82 break;
83
84 case HCSR_TRIGGER:
85 /* Iniciar pulso de trigger: poner TRIGGER en alto */
86 setting->trigger_write(1);
87 setting->t_ref = setting->get_us();
88 setting->state = HCSR_TRIG_WAIT;
89 break;
90
91 case HCSR_TRIG_WAIT:
92 /* Mantener TRIGGER alto al menos 10 us */
93 if ((setting->get_us() - setting->t_ref) >= 10) {
94 setting->trigger_write(0);
95 setting->t_ref = setting->get_us();
96 setting->state = HCSR_ECHO_WH;
97 }
98 break;
99
100 case HCSR_ECHO_WH:
101 /* Esperar flanco ascendente del pin ECHO */
102 if (setting->echo_read() == 1) {
103 setting->t_ref = setting->get_us();
104 setting->state = HCSR_ECHO_WL;
105 } else if ((setting->get_us() - setting->t_ref) >= setting->timeout) {
106 setting->state = HCSR_ERROR;
107 }
108 break;
109
110 case HCSR_ECHO_WL:
111 /* Esperar flanco descendente del pin ECHO y calcular distancia */
112 if (setting->echo_read() == 0) {
113 uint32_t time_echo = setting->get_us() - setting->t_ref;
114 setting->distance = (uint16_t)(time_echo / 58);
115 setting->state = HCSR_READY;
116 }
117 break;
118
119 case HCSR_READY:
120 /* Medicion completa, esperando que el usuario lea el resultado */
121 break;
122
123 case HCSR_ERROR:
124 /* Error de timeout, esperando reset del usuario */
125 break;
126
127 default:
128 setting->state = HCSR_IDLE;
129 break;
130 }
131}
