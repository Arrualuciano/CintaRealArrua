1/**
2 * @file Classifier.h
3 * @brief Clasificador de cajas para cinta transportadora (modo open-loop).
4 *
5 * @details Mantiene una lista de cajas en transito. Cada caja tiene asignado
6 * un servo destino y un timer de arribo (timePushBox). El timer se
7 * decrementa cada 2 ms en Classifier_Process(); al llegar a cero se
8 * dispara el servo correspondiente.
9 *
10 * Flujo de uso:
11 * @code
12 * Classifier_Init(&clas, boxTypes, onTrigger);
13 * Classifier_NewBox(&clas, tipo); // al medir una caja
14 * Classifier_Process(&clas); // cada 2 ms
15 * @endcode
16 *
17 * @author Luciano
18 * @date Mayo 2026
19 */

36/**

57typedef struct {
